1.) Install compiler:

sudo add-apt-repository ppa:ubuntu-toolchain-r/test

sudo apt update

sudo apt-get install gcc g++

sudo apt-get install gcc-9 g++-9

sudo update-alternatives --install /usr/bin/gcc gcc /usr/bin/gcc-9 900 --slave /usr/bin/g++ g++ /usr/bin/g++-9

sudo update-alternatives --install /usr/bin/gcc gcc /usr/bin/gcc-7 700 --slave /usr/bin/g++ g++ /usr/bin/g++-7

2.) Install Cmake:

sudo snap install cmake

or from sources: ( see https://vitux.com/how-to-install-cmake-on-ubuntu-18-04/ )

sudo apt-get install libssl-dev

cd /opt

sudo wget https://github.com/Kitware/CMake/releases/download/v3.14.7/cmake-3.14.7.tar.gz

sudo tar -zxvf cmake-3.14.7.tar.gz

cd cmake-3.14.7

sudo ./bootstrap

sudo make

/usr/bin/cmake --version

next step only if [/usr/bin/cmake] exists and not a symlink

sudo mv /usr/bin/cmake /usr/bin/cmake-3.??.? [specify your version]

sudo make install

#sudo ln -s /opt/cmake-3.14.7/cmake/bin/cmake /usr/bin/cmake )

3.) Prerequistes

sudo apt-get install libgtk-3-dev pkg-config

sudo apt-get install curl unzip tar

sudo apt install python3-pip

sudo pip3 install numpy

sudp ln -s /usr/bin/python3 /usr/bin/python

3.) install vcpkg (C++ package manager)

cd /

sudo mkdir vcpkg

sudo chmod 777 vcpkg

es empfehlen sich kurze Pfade , z.B. von / aus

git clone https://github.com/microsoft/vcpkg.git

cd vcpkg

sudo ./bootstrap-vcpkg.sh

./vcpkg integrate install

4.)optional: [only for qt5]

sudo apt-get install libxi-dev libgl1-mesa-dev libglu1-mesa-dev mesa-common-dev libxrandr-dev libxxf86vm-dev

5.) Install boost , magic-enum und opencv with vcpkg

sudo ./vcpkg install magic-enum

sudo ./vcpkg install libuv

sudo ./vcpkg install boost

sudo ./vcpkg install opencv

optional:

sudo ./vcpkg install qt5

6.) Install source code

git clone https://github.com/zweistein-frm2/Neu2d_det.git

You should have the following directory structure:

|-----\asio-extensions

|

|-----\charm

|

|-----\sigslot

7.) Jetzt braucht man den C++ 17 compiler (gcc 9):

sudo update-alternatives -- config gcc

choose 0

Jetzt braucht man die Pfade aus 3.) die dann in die command line richtig eingefügt werden (also bei euch /home/alanghof oder users/alanghof mit dem bei euch vorhanden Pfad ersetzen)

mkdir hiersolleshin

cd Hiersolleshin

cmake -DCMAKE_TOOLCHAIN_FILE=/vcpkg/scripts/buildsystems/vcpkg.cmake -DSIGSLOT_PATH=/home/alanghof/repos/Neu2d_det/sigslot/include -DASIO_EXTENSIONS_PATH=/home/alanghof/repos/asio-extensions/include -S /home/alanghof/repos/Neu2d_det/charm/ -B /home/alanghof/hiersolleshin/
